from songdl.metadata import lookup_all, clean_youtube_title

title = clean_youtube_title('Rick Astley - Never Gonna Give You Up (Official Video)')

t0 = time.time()
try:
    merged, found = lookup_all(title, 'Rick Astley', 'Rick Astley - Never Gonna Give You Up', {})
